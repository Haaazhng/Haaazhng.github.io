应用寄存器堆设计FIFO队列

FIFO

定义
FIFO队列，即First in first out，与Last in first out 的LIFO栈相对。
特点
没有外部读写地址线，数据由内部读写指针自动加一完成。
分类
分为同步FIFO和异步FIFO。同步FIFO的读与写使用同一个时钟，常用于同步时钟的数据缓存。异步FIFO的读与写相互独立，用于跨时钟的数据信号的传递。

端口说明

以下为FIFO模块的框图：

clk,rst：时钟与复位信号
deq,enq：出入队使能信号，高电平有效，一次即允许操作一项数据
out,in：出/入队数据
full,empty：队满/空信号
an,seg：数码管控制信号，显示队列数据

逻辑结构

采用结构化的方式描述和实现电路。将电路分为队列控制单元LCU、寄存器堆RF和数码管显示单元SDU。内部信号valid[7:0]指示对应寄存器数据是否有效，即是否在队列中。

LCU设计

我们对一些核心代码进行分析。
首先我们需要对deq和enq信号进行处理，代码如下，分别使用三个寄存器来获得有效信号（高电平）只延续单个时钟周期长度的de_p，en_p。如此处理既减少了亚稳态的出现，又保证了出入队信号不会同时有效（互斥），且时状态机的实现更为简单。

always @(posedge clk) begin         //出入队使能信号处理
    de_st<=deq;
    en_st<=enq;
    de_s<=de_st;
    en_s<=en_st;
    de_pt<=de_s;
    en_pt<=en_s;
end
assign de_p=de_s&(~de_pt),en_p=en_s&(~en_pt);

然后处理队满/空信号：

always @(*) begin
    if(valid==0) empty=1;           //队空
    else empty=0;
    if(wp==rp && valid != 0) full=1;    //队满
    else full=0;
end

当valid的值为0时，说明无有效数据，即队列空，此时wp=rp；而当读指针rp与写指针wp值相同，且valid不为零时，说明队列满。

最后时三段式有限状态机的实现：、

always @(posedge clk) begin         //时序逻辑
    if(rst) begin
        curr_state<=IDLE;
    end
    else curr_state<=next_state;
end

always @(*) begin
    if(de_p)begin
        if(empty==0) next_state=DEQU;       //出队信号有效且队列不空
        else next_state=IDLE;           //队列空
    end
    else if(en_p)begin
        if(full==0) next_state=ENQU;        //入队信号有效且队列未满
        else next_state=IDLE;               //队列满
    end
    else next_state=IDLE;
end

always @(*) begin       //根据状态更新写/读使能信号
    case(curr_state)
    ENQU:begin
        we=1;
        re0=0;
    end
    DEQU:begin
        we=0;
        re0=1;
    end
    IDLE:begin
        we=0;
        re0=0;
    end
    default:begin
        we=0;
        re0=0;
    end
    endcase
end

assign wd=in,wa=wp;
assign ra0=rp,out=rd0;

为了方便描述与控制，我们给寄存器堆增加了读使能信号re。在状态机第三部分，根据当前状态值来令给读使能信号re0和写使能信号we赋值，从而实现对寄存器的读取与写入。

对于读写指针，我们需要使指针能够从0~7循环移动。直接给出LCU完整代码：

module LCU(
    input clk, rst,
    input [3:0] in,
    input enq,
    input deq,
    input [3:0] rd0,
    output [3:0] out,
    output reg empty,full,
    output [2:0] wa,
    output [2:0] ra0,
    output [3:0] wd,
    output reg we,re0,
    output reg [7:0] valid
    );
    parameter IDLE = 0 , ENQU = 1 , DEQU = 2;   //状态值
    reg [1:0] curr_state,next_state;
    reg de_st,de_s,de_pt,en_st,en_s,en_pt;
    wire de_p,en_p;                             //出入队信号处理
    reg [2:0] wp,rp,wp_ns,rp_ns;

    always @(posedge clk) begin         //时序逻辑
        if(rst) begin
            curr_state<=IDLE;
        end
        else curr_state<=next_state;
    end

    always @(*) begin
        if(de_p)begin
            if(empty==0) next_state=DEQU;       //出队信号有效且队列不空
            else next_state=IDLE;           //队列空
        end
        else if(en_p)begin
            if(full==0) next_state=ENQU;        //入队信号有效且队列未满
            else next_state=IDLE;               //队列满
        end
        else next_state=IDLE;
    end


    always @(posedge clk) begin         //更新指针
        if(rst) begin
            wp<=0;
            rp<=0;
        end
        else begin
            wp<=wp_ns;
            rp<=rp_ns;
        end
    end

    always @(*) begin
        if(curr_state==ENQU)begin       //将入队时，更新写指针
        case(wp)
            0:begin wp_ns=1; end
            1:begin wp_ns=2; end
            2:begin wp_ns=3; end
            3:begin wp_ns=4; end
            4:begin wp_ns=5; end
            5:begin wp_ns=6; end
            6:begin wp_ns=7; end
            7:begin wp_ns=0; end
        endcase
        end
        else wp_ns=wp;
        if(curr_state==DEQU)begin       //将出队时，更新读指针
        case(rp)
            0:begin rp_ns=1; end
            1:begin rp_ns=2; end
            2:begin rp_ns=3; end
            3:begin rp_ns=4; end
            4:begin rp_ns=5; end
            5:begin rp_ns=6; end
            6:begin rp_ns=7; end 
            7:begin rp_ns=0; end
        endcase     
        end
        else rp_ns=rp;
    end

    always @(*) begin       //根据状态更新写/读使能信号
        case(curr_state)
        ENQU:begin
            we=1;
            re0=0;
        end
        DEQU:begin
            we=0;
            re0=1;
        end
        IDLE:begin
            we=0;
            re0=0;
        end
        default:begin
            we=0;
            re0=0;
        end
        endcase
    end

    assign wd=in,wa=wp;
    assign ra0=rp,out=rd0;

    always @(posedge clk) begin     //更改valid对应值
        if(curr_state==DEQU)
            valid[rp]<=0;
        if(curr_state==ENQU)
            valid[wp]<=1;
        if(rst)
            valid<=0;
    end
    always @(*) begin
        if(valid==0) empty=1;           //队空
        else empty=0;
        if(wp==rp && valid != 0) full=1;    //队满
        else full=0;
    end
    always @(posedge clk) begin         //出入队使能信号处理
        de_st<=deq;
        en_st<=enq;
        de_s<=de_st;
        en_s<=en_st;
        de_pt<=de_s;
        en_pt<=en_s;
    end
    assign de_p=de_s&(~de_pt),en_p=en_s&(~en_pt);

endmodule

RF设计

re0和re1是我们为了方便LCU模块的实现添加的两个读使能端口。读端口2连接在SDU模块，因此我们需要在顶层模块中令re1恒为1。请注意，由于寄存器堆的x0寄存器硬编程值为0，因此当该寄存器在队列中时，写入的数据始终无效。

module RF             //三端口32 xWIDTH寄存器堆
(   
    input clk,                      //时钟（上升沿有效）
    input [2 : 0] ra0,               //读端口0地址
    output reg [ 3 : 0] rd0,      //读端口0数据
    input [2: 0] ra1,                //读端口1地址
    output reg [3 : 0] rd1,      //读端口1数据
    input [2 : 0] wa,                //写端口地址
    input we,                       //写使能，高电平有效
    input re0,re1,
    input [3 : 0] wd         //写端口数据
);
    reg [3 : 0] regfile[0 : 7];

    always @ (posedge clk) begin
        regfile[0]<=0;
        if(re0)begin
            rd0 = regfile[ra0];
        end
        if(re1)begin
            rd1 = regfile[ra1];
        end
        if (we==1 && wa!=0) regfile[wa] <= wd;
    end
endmodule

SDU设计

该模块的功能是将队列中的有效数据输出在八个数码管中的对应位置。即将valid值不为0的寄存器数据（也即队列数据，队列数据则是存储在寄存器堆中的）输出在数码管。这里使用了时分复用的方式依次显示。

module SDU (
    input clk,
    input [3:0] rd1,
    input [7:0] valid,
    output reg [2:0] ra1,
    output reg [2:0] an,
    output [3:0] seg
);
    reg [15:0] cnt=0;       //分时复用计数
    always @(posedge clk) begin
        cnt<=cnt+1;
    end

    always @(posedge clk) begin
        if(valid==0)begin       //队空时，显示第一个元素（0）
            an<=0;
            ra1<=0;
        end
        else begin          //不为空时
            case(cnt[15:13])
            0:begin
                if(valid[0])begin   //显示队列中的元素
                    ra1<=0;
                    an<=0;
                end
            end
            1:begin
                if(valid[1])begin
                    ra1<=1;
                    an<=1;
                end
            end
            2:begin
                if(valid[2])begin
                    ra1<=2;
                    an<=2;
                end
            end
            3:begin
                if(valid[3])begin
                    ra1<=3;
                    an<=3;
                end
            end
            4:begin
                if(valid[4])begin
                    ra1<=4;
                    an<=4;
                end
            end
            5:begin
                if(valid[5])begin
                    ra1<=5;
                    an<=5;
                end
            end
            6:begin
                if(valid[6])begin
                    ra1<=6;
                    an<=6;
                end
            end
            7:begin
                if(valid[7])begin
                    ra1<=7;
                    an<=7;
                end
            end
            endcase
        end
    end
    assign seg=rd1;
endmodule

Top模块

由于我们使用了结构化方式实现电路，因此我们只需要在顶层模块中建立好内部变量，并分别例化LCU、SDU、RF模块即可。

module fifo (
    input clk, rst,         //时钟（上升沿有效）、同步复位（高电平有效）
    input enq,              //入队列使能，高电平有效
    input [3:0] in,         //入队列数据
    input deq,              //出队列使能，高电平有效
    output [3:0] out,       //出队列数据
    output full,empty,      //队列满和空标志
    output [2:0] an,        //数码管选择
    output [3:0] seg        //数码管数据
);
    wire [2:0] ra0,ra1,wa;
    wire [3:0] rd0,rd1,wd;
    wire we,re0,re1;
    wire [7:0] valid; 

    assign re1=1;
    LCU lcu_inst(
        .rst(rst),
        .clk(clk),
        .enq(enq),
        .deq(deq),
        .empty(empty),
        .full(full),
        .in(in),
        .out(out),
        .we(we),
        .re0(re0),
        .rd0(rd0),
        .ra0(ra0),
        .wd(wd),
        .wa(wa),
        .valid(valid)
    );

    SDU sdu_inst(
        .clk(clk),
        .rd1(rd1),
        .ra1(ra1),
        .an(an),
        .seg(seg),
        .valid(valid)
    );

    RF rf_inst(
        .clk(clk),
        .ra0(ra0),
        .ra1(ra1),
        .wa(wa),
        .rd0(rd0),
        .rd1(rd1),
        .we(we),
        .re0(re0),
        .re1(re1),
        .wd(wd)
    );
endmodule

结束

以上便是FIFO信号的完整设计与实现，大家自行编写仿真文件进行仿真和编写约束文件生成bit流烧写在fpga平台。其实Vivado中已经给我们提供众多的ip核，相当于Vivado已经给我们写好了很多常用或复杂的模块，我们需要时只需要找到对应的ip核并对其例化即可。其中就包括了分布式存储器和块式存储器，大家可尝试通过例化ip核来代替RF模块。
